int main()
{
	 float x0, x1, x2, f0, f1, f2, e;
	 int step = 1;

	 /* Inputs */
	 up:
	 printf("\nEnter two initial guesses:\n");

	 printf("Enter tolerable error:\n");
	 scanf("%f", &e);
	 /* Calculating Functional Value */
	 f0 = f(x0);
	 f1 = f(x1);
	 /* Checking whether given guesses brackets the root or not. */

	 {
		  printf("Incorrect Initial Guesses.\n");
		  goto up;
	 }
   /* Implementing Bisection Method */
	 printf("\nStep\t\tx0\t\tx1\t\tx2\t\tf(x2)\n");
	 do
	 {
		  x2 = (x0 + x1)/2;
		  f2 = f(x2);
		
		  printf("%d\t\t%f\t%f\t%f\t%f\n",step, x0, x1, x2, f2);
		
		  if( f0 * f2 < 0)
		  {
			   x1 = x2;
			   f1 = f2;
		  }
		  else
		  {
			   x0 = x2;
			   f0 = f2;
		  }
		  step = step + 1;

#include <stdio.h>
#include <math.h>

int fact(int);

void main() {
    float arr[10][11], x, h, p, y, px = 1;
    int i, j, n, ch = 30;

    printf("\nEnter the number of data:");
    scanf("%d", &n);

    printf("\nEnter the data");
    for (i = 0; i < n; i++) {
        printf("X%d=", i + 1);
        scanf("%f", &arr[i][0]);
        printf("Y%d=", i + 1);
        scanf("%f", &arr[i][1]);
    }

    // Forming difference table.
    for (j = 2; j <= n; j++)
        for (i = 0; i < n - 1; i++)
            arr[i][j] = arr[i + 1][j - 1] - arr[i][j - 1];

    // Printing table.
    printf("\nDifference table is:-");
    printf("\n\tx\tY");
    for (i = 0; i <= n - 2; i++)
        printf("\t%c^%dY", ch, i + 1);
    for (i = 0; i < n; i++) {
        printf("\n");
        for (j = 0; j < n + 1 - i; j++) {
            printf("\t%.4f", arr[i][j]);
        }
    }

    // Take the value of x for f(x).
    printf("\nEnter the value x for function f(x):");
    scanf("%f", &x);

    // Calculate the value of f(x) for x.
    h = arr[1][0] - arr[0][0];
    p = (x - arr[0][0]) / h;
    y = arr[0][1];

    for (i = 1; i < n; i++) {
        px = px * (p - (i - 1));
        y = y + (arr[0][i + 1] * px) / fact(i);
    }

    printf("\nThe value of function at x=%f is %f", x, y);
}

int fact(int n) {
    int i, f = 1;
    for (i = 1; i <= n; i++)
        f = f * i;
    return f;
}

#include <stdio.h>
#include <math.h>

int fact(int);

void main() {
    float arr[10][11], x, h, p, y, px = 1;
    int i, j, n, ch = 30;

    printf("\nEnter the number of data:");
    scanf("%d", &n);

    printf("\nEnter the data");
    for (i = 0; i < n; i++) {
        printf("X%d=", i + 1);
        scanf("%f", &arr[i][0]);
        printf("Y%d=", i + 1);
        scanf("%f", &arr[i][1]);
    }

    // Forming difference table (backward differences).
    for (j = 2; j <= n; j++)
        for (i = n - 1; i >= j - 1; i--)
            arr[i][j] = arr[i][j - 1] - arr[i - 1][j - 1];

    // Printing table.
    printf("\nDifference table is:-");
    printf("\n\tx\tY");
    for (i = 0; i <= n - 2; i++)
        printf("\t%c^%dY", ch, i + 1);
    for (i = 0; i < n; i++) {
        printf("\n");
        for (j = 0; j < n + 1 - i; j++) {
            printf("\t%.4f", arr[i][j]);
        }
    }

    // Take the value of x for f(x).
    printf("\nEnter the value x for function f(x):");
    scanf("%f", &x);

    // Calculate the value of f(x) for x.
    h = arr[1][0] - arr[0][0];
    p = (x - arr[n - 1][0]) / h;
    y = arr[n - 1][1];

    for (i = 1; i < n; i++) {
        px = px * (p + (i - 1));
        y = y + (arr[n - 1][i + 1] * px) / fact(i);
    }

    printf("\nThe value of function at x=%f is %f", x, y);
}

###################Bisection Method Algorithm##############
Follow the below procedure to get the solution for the continuous function:

For any continuous function f(x),

Find two points, say a and b such that a < b and f(a)* f(b) < 0
Find the midpoint of a and b, say â€œtâ€
t is the root of the given function if f(t) = 0; else follow the next step
Divide the interval [a, b] â€“ If f(t)*f(a) <0, there exist a root between t and a
â€“ else if f(t) *f (b) < 0, there exist a root between t and b
Repeat above three steps until f(t) = 0.
The bisection method is an approximation method to find the roots of the given equation by repeatedly dividing the interval. This method will divide the interval until the resulting interval is found, which is extremely small.
